{{- /* cite — inline parenthetical citation (author year, pages). Looks up the key in hugo.Data.bibliography (produced by scripts/build_bibliography.py) and emits a compact author-date note that also links to the full entry on /sources/#{key}. Usage: {{< cite key="cawley1993federal" >}} {{< cite key="cawley1993federal" pages="42-45" >}} {{< cite author="Cawley" year="1993" pages="42-45" >}} (manual, no key) */ -}} {{- $key := .Get "key" -}} {{- $author := .Get "author" -}} {{- $year := .Get "year" -}} {{- $pages := .Get "pages" -}} {{- $entry := "" -}} {{- if $key -}} {{- $entry = index hugo.Data.bibliography $key -}} {{- end -}} {{- if $entry -}} {{- if not $author }}{{ $author = $entry.authorlast }}{{ end -}} {{- if not $year }}{{ $year = $entry.year }}{{ end -}} {{- end -}} ({{ with $author }}{{ . }}{{ end }}{{ with $year }} {{ . }}{{ end }}{{ with $pages }}, {{ . }}{{ end }})
