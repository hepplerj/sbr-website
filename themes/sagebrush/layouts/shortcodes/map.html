{{- /* map — render an interactive Leaflet map inline within narrative prose. For full-bleed standalone map pages, use frontmatter .Params.map and the layouts/maps/single.html layout instead. Required: data Path to a GeoJSON file Optional (all lowercase — Hugo lowercases frontmatter keys, we follow suit): id DOM id for the figure height Pixel height (default 500) center "lat,lon" zoom Initial zoom (default 5) caption Markdown caption rendered under the map style Palette preset: sage (default), green, gold, rust, navy valuefield Feature property for the choropleth ramp + info/popup value labelfield Feature property for display label valueunit Unit suffix (e.g. "%", " acres") legendtitle Title for the legend infotitle Hover-panel default title infoprompt Hover-panel default prompt text popupnote Small note under value in click popups */ -}} {{- $id := .Get "id" | default (printf "map-%d" .Ordinal) -}} {{- $data := .Get "data" -}} {{- $height := .Get "height" | default "500" -}} {{- $caption := .Get "caption" -}} {{- $center := .Get "center" -}} {{- $zoom := .Get "zoom" | default "5" -}} {{- $style := .Get "style" | default "sage" -}} {{- $layer := dict "src" $data "style" $style -}} {{- with .Get "labelfield" }}{{ $layer = merge $layer (dict "labelfield" .) }}{{ end -}} {{- with .Get "valuefield" }}{{ $layer = merge $layer (dict "valuefield" .
